#include<stdio.h>
#include<windows.h>
#include<unistd.h>


void ifte(int a,int b)
{
    COORD position;
    position.X = a;
    position.Y = b;
    SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE),position);
}

int main()

{


    int i,xa,ya;
    int time= 200000;

    while(1)
    {

    for(i=5;i<=30;i+=1)
    {
        int j = 15;

        ///loop for I:

        //* for I upper horizontal part:
        ifte(i,j);
        for(int k=1;k<=5;k+=1)
            printf("*");
        fflush(stdout);

        //* for I vertical part:
        xa = i+2;
        ya = j+1;
        for(int k=0;k<3;k+=1)
        {
            ifte(xa,ya+k);
            printf("*");
        }
        fflush(stdout);

        //* for I lower horizontal part:
        ifte(i,j+4);
        for(int k=1;k<=5;k+=1)
            printf("*");
        fflush(stdout);

        //space for I upper horizontal part(forward I):
        ifte(i-5,j);
        for(int k=1;k<=5;k+=1)
            printf(" ");
        fflush(stdout);

        //space for I  vertical part:
        xa = i + 1;
        ya = j + 1;
        for(int k=0;k<3;k+=1)
        {
            ifte(xa,ya+k);
            printf(" ");
        }
        fflush(stdout);

        //space for I lower horizontal part(forward I):
        ifte(i-5,j+4);
        for(int k=1;k<=5;k+=1)
            printf(" ");
        fflush(stdout);





        ///loop for F:
        //space for F vertical part
        xa = i + 5;
        for(int k=0;k<5;k+=1)
        {
            ifte(xa,j+k);
            printf(" ");
        }
        fflush(stdout);

        //* for F upper horizontal part
        ya = j;
        ifte(i+6,ya);
        for(int k=1;k<=5;k+=1)
            printf("*");
        fflush(stdout);

        //* for F middle horizontal part
        ifte(i+6,j+2);
        for(int k=1;k<=5;k+=1)
            printf("*");
        fflush(stdout);

        //* for F vertical part
        xa = i + 6;
        for(int k=0;k<5;k+=1)
        {
            ifte(xa,j+k);
            printf("*");
        }
        fflush(stdout);



        ///loop for T:
        //* for T upper horizontal part
        ifte(i+12,j);
        for(int k=1;k<=5;k+=1)
            printf("*");
        fflush(stdout);

        //* for T vertical part
        xa = i + 14;
        for(int k=1;k<5;k+=1)
        {
            ifte(xa,j+k);
            printf("*");
        }
        fflush(stdout);

        //space for T upper horizontal part(forward T)
        ifte(i+11,j);
        printf(" ");
        fflush(stdout);

        //space for T vertical part
        xa = i + 13;
        for(int k=1;k<=5;k+=1)
        {
            ifte(xa,j+k);
            printf(" ");
        }
        fflush(stdout);




        ///loop for E:

        //* for E upper horizontal part
        ifte(i+18,j);
        for(int k=1;k<=5;k+=1)
            printf("*");
        fflush(stdout);

        //* for E middle horizontal part
        ifte(i+18,j+2);
        for(int k=1;k<=5;k+=1)
            printf("*");
        fflush(stdout);

        //* for E lower horizontal part
        ifte(i+18,j+4);
        for(int k=1;k<=5;k+=1)
            printf("*");
        fflush(stdout);

        //* for E vertical part
        xa = i + 18;
        for(int k=0;k<5;k+=1)
        {
            ifte(xa,j+k);
            printf("*");
        }
        fflush(stdout);

        //space for E vertical part
        xa = i + 17;
        for(int k=0;k<5;k+=1)
        {
            ifte(xa,j+k);
            printf(" ");
        }
        fflush(stdout);

        usleep(time);
    }
    printf("\033[2J");
}


    getch();
}
